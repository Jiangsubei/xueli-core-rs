### 3.4 EmojiManager：修复分类循环启动

**文件**：`src/emoji/manager.rs:107-116`

**现状**：`start_idle_classification_loop_from_init()` 为空存根

**方案**：在 `initialize()` 中 spawn 后台任务：

```rust
pub async fn initialize(&self) {
    // ... 现有初始化逻辑 ...
    self.start_classification_loop().await;
}

async fn start_classification_loop(&self) {
    let manager = self.clone();
    tokio::spawn(async move {
        let mut interval = tokio::time::interval(Duration::from_secs(300));
        loop {
            interval.tick().await;
            if !manager.can_run_classification_now() {
                continue;
            }
            if let Err(e) = manager.run_classification_cycle().await {
                warn!("[Emoji] 分类周期失败: {e}");
            }
        }
    });
}
```

```rust
    let memories = self.important_store.get_user_memories(user_id).await?;
    let mut facts = Vec::new();
    for (_, mem) in memories {
        if self.should_use_as_fact(&mem) {
            let fact_kind = self.infer_fact_kind(&mem);
            let fact = PersonFact {
                fact_kind,
                content: mem.content.clone(),
                // ...
            };
            facts.push(fact);
        }
    }
    Ok(facts)
}
```
